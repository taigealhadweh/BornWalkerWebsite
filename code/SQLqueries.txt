--
-- get one user's all goals(including achieved/not) by userid
--
select goal.goal_ID, achievement.achieved, user.userid
from goal, achievement, user
where goal.goal_ID = achievement.goal_ID
and user.userid = achievement.userid
and user.userid = 1;

--
-- get one user's achieved goals by userid
--
select goal.goal_ID, achievement.achieved, user.userid
from goal, achievement, user
where goal.goal_ID = achievement.goal_ID
and user.userid = achievement.userid
and user.userid = 1
and achievement.achieved = 'ture';

--
-- get one user's not achieved goals by userid
--
select goal.goal_ID, achievement.achieved, user.userid
from goal, achievement, user
where goal.goal_ID = achievement.goal_ID
and user.userid = achievement.userid
and user.userid = 1
and achievement.achieved = 'false';
